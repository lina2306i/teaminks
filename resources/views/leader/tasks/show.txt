    function dateBadge($dueDate) {
        if (!$dueDate) return null;

        $daysLeft = now()->diffInDays($dueDate, false);

        return [
            'daysLeft'   => $daysLeft,
            'isOverdue'  => $daysLeft < 0,
            'isToday'    => $daysLeft === 0,
            'isTomorrow' => $daysLeft === 1,
            'isSoon'     => $daysLeft > 1 && $daysLeft <= 2,
        ];
    }
